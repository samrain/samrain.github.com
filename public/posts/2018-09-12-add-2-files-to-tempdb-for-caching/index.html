给tempdb加2个文件做缓存

sql脚本如下：

USE [master];
GO
alter database tempdb modify file (name='tempdev', size = 8GB);
GO
/* Adding three additional files */
USE [master];
GO
ALTER DATABASE [tempdb] ADD FILE (NAME = N'tempdev2', FILENAME = N'E:\SQLDATA\tempdev2.ndf' , SIZE = 8GB , FILEGROWTH = 0);
GO
